## 메타 태그 

```html
    <!-- 캐릭터 셋은 utf-8 -->
    <meta charset="UTF-8">

    <!-- 모바일기기의 가로 사이즈에 화면 사이즈를 맞춰줌-->
    <meta name="viewport" content="width=device-width, initial-scale=1, shrink-to-fit=no" >
    <!-- SNS 공유 -->
    <meta property="og:type" content="website">
    <meta property="og:title" content="웹사이트 타이틀" />
    <meta property="og:description" content="웹사이트 설명 " />
    <meta property="og:image" content="웹사이트 대표 이미지 " />
    <meta property="og:url" content="웹사이트주소" />

    <!-- 트위터 카드 타입은 summary_large_image, summary, photo 중 하나를 선택할 수 있다. -->
    <meta property="twitter:card" content="summary" />
    <meta name="description" content="웹사이트 설명 ">
    <meta name="twitter:url" content="트위터카드 클릭시 이동하는 URL ">
    <meta name="twitter:title" content="트위터 카드에 나오는 타이틀 ">
    <meta name="twitter:description" content="트위터 카드에 나오는 설명 ">
    <meta name="twitter:image" content="트위터 카드에 뜨는 이미지 ">
    <meta name="author" content="작성자 이름 혹은 email">
```
